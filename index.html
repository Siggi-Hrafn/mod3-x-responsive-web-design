Container / Wrapper

CSS - .container-1 - min() function


.container-1 {
/* min function, 2 arg = min-width 100% & max-width 95rem=950px */
width: min(100%, 95rem);
/* margin on left and right */
margin-inline: auto;
/* border: width, style, color; */
border: 0.3rem solid red;
/* padding 3.2rem=32px after conversion */
padding: 3.2rem;
}

CSS - .container-2 - clamp() function

.container-2 {
/*clamp(min, val, max) => min-with 32rem = 320px, val = 90% of viewport width, max-width= 144rem=1440px */
width: clamp(32rem, 90vw, 144rem);
/* margin on left and right */
margin-inline: auto;
/* border: width, style, color; */
border: 0.3rem solid green;
/* padding 3.2rem=32px after conversion */
padding: 3.2rem;
}

}
}

.text-columns {
column-count: 3;
column-gap: 8rem;
}
}
